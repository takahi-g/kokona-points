function editLog(id) { const index = logs.findIndex(l => l.id === id); const currentEntry = logs[index]; // 日付と内容をセットで聞くようにしたよ const newDate = prompt("日付をなおす？ (例: 1/16)", currentEntry.day); if (newDate === null) return; // キャンセルされたら終わり const newText = prompt("内容をなおすよ：", currentEntry.text); if (newText === null) return; // キャンセルされたら終わり if (newDate !== "" && newText !== "") { logs[index].day = newDate; logs[index].text = newText; saveData(); updateLogDisplay(); } }
